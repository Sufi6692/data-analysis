create database bank;

use bank;

--create a customer table
CREATE TABLE customer (
cnum INT NOT NULL,
fname VARCHAR(30) NOT NULL,
lname VARCHAR(30) NOT NULL,
addr VARCHAR(100) NOT NULL,
city VARCHAR(100) NOT NULL,
contact VARCHAR(30) NOT NULL,
pincode int NOT NULL,
PRIMARY KEY (cnum)
);

--create a account table
CREATE TABLE account (
account_id VARCHAR(30) NOT NULL,
cnum INT NOT NULL,
type VARCHAR(30) NOT NULL,
balance decimal(10,2) NOT NULL,
PRIMARY KEY (account_id),
FOREIGN KEY (cnum) REFERENCES customer(cnum)
);

--create a transaction table
CREATE TABLE TXN (  
txn_id int identity,
account_id VARCHAR(30) NOT NULL,
txn_dt date NOT NULL,
channel VARCHAR(30) NOT NULL,
amount decimal(10,2) NOT NULL,
PRIMARY KEY (txn_id),
FOREIGN KEY (account_id) REFERENCES account(account_id)
);

--before adding any data in txn table create triggers for insert, update and delete





--create insert trigger
CREATE TRIGGER update_balance ON txn  
AFTER INSERT  
AS  
BEGIN  
	SET NOCOUNT ON; -- it will not return total records affected messages
	declare @amount decimal(10,2); -- local variables
	declare @account_id varchar(30);

	select @account_id=account_id, @amount=amount from inserted; --inserted is a system table which will have last inserted data
	

	update account 
	set balance = balance + @amount
	where account.account_id = @account_id;
END;

--trigger after update
CREATE TRIGGER update_balance_2 ON txn  
AFTER UPDATE  
AS  
BEGIN  
	SET NOCOUNT ON;
	declare @amount decimal(10,2);
	declare @account_id varchar(30);
	declare @old_amount decimal(10,2);
	declare @old_account_id varchar(30);


	select @old_account_id=account_id, @old_amount=amount from deleted

	--reverse the old txn
	update account 
	set balance = balance - @old_amount
	where account.account_id = @old_account_id;

	select @account_id=account_id, @amount=amount from inserted; 
	
	update account 
	set balance = balance + @amount
	where account.account_id = @account_id;
 
END;  


--modify existing transaction to see the balance being updated

update txn
set amount = 35000
where txn_id = 1;

--trigger after delete
CREATE TRIGGER update_balance_3 ON txn  
AFTER DELETE
AS  
BEGIN  
	SET NOCOUNT ON;
	declare @old_amount decimal(10,2);
	declare @old_account_id varchar(30);

	select @old_account_id=account_id, @old_amount=amount from deleted

	update account 
	set balance = balance - @old_amount
	where account.account_id = @old_account_id;

END;
