--find list of orders for customers who live in London

select * from orders;

--join query
select onum, amt, odate, c.cnum, cname from 
orders o join customer c
on
o.cnum = c.cnum
where city = 'London';

--subquery
select * from orders where
cnum in (select cnum from customer where city = 'London');

select * from orders where

--find customer whose avg order value is above 2000
select * from customer where cnum in (
select cnum from orders
group by cnum
having avg(amt) > 2000);

--Rank function
CREATE TABLE rank_table (Name VARCHAR(10));

INSERT INTO rank_table (Name) 
 VALUES('A'), ('C'), ('C'), ('B'), ('B'), ('D'), ('E');

select * from rank_table;

SELECT Name, 
RANK () OVER (
ORDER BY Name
) AS Rank_no 
FROM rank_table;

--find 2nd rank 
select * from 
(SELECT Name, 
RANK () OVER (
ORDER BY Name
) AS Rank_no 
FROM rank_table) t1
where rank_no = 2;

--FIND DENSE_RANK()
SELECT Name, 
DENSE_RANK () OVER (
ORDER BY Name
) AS Rank_no 
FROM rank_table;

--find top 5
select name, rank_no from (
SELECT Name, 
RANK () OVER (
ORDER BY Name
) AS Rank_no 
FROM rank_table) x where Rank_no <=5 ;
